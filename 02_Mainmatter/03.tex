\begin{equation}\label{eq}

\subsection{Precision}

\begin{equation}\label{eq}
    Precision = \frac{TP}{TP + FP}
\end{equation}

\subsection{F1 Score}

\begin{equation}\label{eq}
    F1 = \frac{2 \times Precision \times Recall}{Precision + Recall} = \frac{2 \times TP}{2 \times TP + FP + FN}
\end{equation}
